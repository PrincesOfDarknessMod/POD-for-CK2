# era changer

character_event = {
	id = POD_cultural_conversion.1
	hide_window = yes	
	is_triggered_only = yes

	trigger = {
		trait = storyteller
	}

	immediate = {
	
		if = {
			limit = { 
				NOT = { has_global_flag = era_ancient }
				NOT = { has_global_flag = era_medieval }
				NOT = { has_global_flag = era_renaissance }
				NOT = { has_global_flag = era_imperial }
				NOT = { has_global_flag = era_industrial }
				NOT = { has_global_flag = era_modern }			
			}	
			set_global_flag = era_medieval
		}
			

		if = {
			limit = { 
				year = 1450
				NOT = { year = 1650 }
				NOT = { has_global_flag = era_ancient }
				has_global_flag = era_medieval
				NOT = { has_global_flag = era_renaissance }
				NOT = { has_global_flag = era_imperial }
				NOT = { has_global_flag = era_industrial }
				NOT = { has_global_flag = era_modern }			
			}
			set_global_flag = era_renaissance	
			clr_global_flag = era_medieval
		}
		
		if = {
			limit = { 
				year = 1650
				NOT = { year = 1850 } 
				has_global_flag = era_renaissance
				NOT = { has_global_flag = era_ancient }
				NOT = { has_global_flag = era_medieval }
				NOT = { has_global_flag = era_imperial }
				NOT = { has_global_flag = era_industrial }
				NOT = { has_global_flag = era_modern }			
			}
			set_global_flag = era_imperial
			clr_global_flag = era_renaissance			
		}		
		
		if = {
			limit = { 
				year = 1850
				NOT = { year = 1900 }
				has_global_flag = era_imperial
				NOT = { has_global_flag = era_ancient }
				NOT = { has_global_flag = era_medieval }
				NOT = { has_global_flag = era_renaissance }
				NOT = { has_global_flag = era_industrial }
				NOT = { has_global_flag = era_modern }			
			}
			set_global_flag = era_industrial
			clr_global_flag = era_imperial
		}	

		if = {
			limit = { 
				year = 1900
				NOT = { year = 2077 }
				has_global_flag = era_industrial
				NOT = { has_global_flag = era_ancient }
				NOT = { has_global_flag = era_medieval }
				NOT = { has_global_flag = era_renaissance }
				NOT = { has_global_flag = era_imperial }
				NOT = { has_global_flag = era_modern }			
			}
			set_global_flag = era_modern
			clr_global_flag = era_industrial
		}			
	
	}
}
