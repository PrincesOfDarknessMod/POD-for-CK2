spawn_anthony_of_padua = {
	if = {
		limit = { 
			year = 1233
			NOT = { year > 1233 }
		}
		random_province = {
			limit = {
				culture = "portuguese"
			}
			create_character = {
				name = "Fernando Martins"
				age = 35
				dna="bfdca0i0aa0"
				properties="fc0c0k000000000000"
				trait = mortal
				trait = canon
				trait = great_person
				trait = grey_eminence
				trait = pod_saint_trait
				trait = monk
				trait = theologian
				trait = erudite
				trait = honest
				trait = humble
				trait = zealous
				female = no
				religion = innocent
				dynasty = none
				culture = "portuguese"
				attributes = {
					diplomacy = 17
					learning = 15
				}
			}
			new_character = {
				save_event_target_as = great_person
				set_character_flag = great_person
				set_character_flag = anthony_of_padua
				set_character_flag = great_theologian
				pod_spawn_great_person_effect = yes
			}
		}
	}
}
